# 10816 카드게임2
input의 범위를 유의해서 봐야하는 문제
```
#define MAXN 10000010
#include <cstdio>
#include <iostream>
using namespace std;

int card1[MAXN],card2[MAXN];
int main(void){
    int N,M,buf;
    const int offset = 10000000;
    cin>>N;
    for(int i=0;i<N;i++){
        scanf("%d",&buf);
        if(buf>=0)  card1[buf]++;
        else    card2[buf+offset]++;
    }
    cin>>M;
    for(int i=0;i<M;i++){
        scanf("%d",&buf);
        if(buf>=0)  printf("%d ",card1[buf]);
        else    printf("%d ",card2[buf+offset]);
    }
    return 0;
}
```
